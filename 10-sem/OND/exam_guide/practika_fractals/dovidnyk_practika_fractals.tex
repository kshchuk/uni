\section{Обов'язкові та додаткові задачі (тема~4)}
% =====================================================================

Для самоподібних множин (губка Менгера, килим Серпінського тощо) фрактальна
(ємнісна) розмірність збігається з розмірністю Хаусдорфа.

\subsection{Алгоритм знаходження фрактальної розмірності}

Якщо множина складається з $N$ копій самої себе, кожна зменшена в $k$ разів
(коефіцієнт подібності $r=1/k$), то
\begin{equation}
  D=\frac{\ln N}{\ln k}.
  \label{eq:fractal-dim}
\end{equation}

\begin{enumerate}[leftmargin=1.6em]
  \item \textbf{Ітераційний крок:} на кожному кроці вихідну фігуру (відрізок,
        квадрат, куб) розбивають на рівні частини.
  \item \textbf{Число копій $N$:} підрахувати, скільки зменшених копій залишається
        після видалення «зайвих» частин.
  \item \textbf{Масштаб $k$:} визначити, у скільки разів лінійний розмір копії
        менший за розмір вихідної фігури.
  \item \textbf{Обчислення:} підставити $N$ і $k$ у формулу~\eqref{eq:fractal-dim}.
\end{enumerate}

\subsection{Задача 4.1. Фрактальна розмірність губки Менгера}

\begin{korotko}
Губка Менгера~--- тривимірний аналог килима Серпінського: куб $3\times3\times3$
дає $N=20$ копій при масштабі $k=3$. Тоді
$D=\ln(20)/\ln(3)\approx 2{,}7268$.
\end{korotko}

Знайдіть фрактальну розмірність губки Менгера.

Куб ділять на $3\times3\times3=27$ менших кубів. Видаляють центральний куб на
кожній з $6$ граней і центральний куб всього великого куба (разом $7$ кубів).
Залишається $N=27-7=20$ копій; лінійний розмір кожної копії в $k=3$ рази
менший за вихідний куб.

\begin{equation*}
  D=\frac{\ln 20}{\ln 3}\approx\frac{2{,}995}{1{,}098}\approx 2{,}7268.
\end{equation*}

\subsection{Задача 4.2. Фрактальна розмірність килима Серпінського}

\begin{korotko}
Квадрат ділять на $3\times3=9$ частин, центральну прибирають: $N=8$, $k=3$.
Отже $D=\ln(8)/\ln(3)\approx 1{,}8928$.
\end{korotko}

Знайдіть фрактальну розмірність килима Серпінського.

Квадрат ділять на $3\times3=9$ менших квадратів; центральний видаляють.
Залишається $N=8$ копій, кожна з лінійним розміром у $k=3$ рази меншим.

\begin{equation*}
  D=\frac{\ln 8}{\ln 3}\approx\frac{2{,}079}{1{,}098}\approx 1{,}8928.
\end{equation*}

Фрактальна розмірність характеризує «заповненість» простору самоподібною
множиною; для класичних фракталів вона є дробовою, на відміну від цілочисельної
євклідової розмірності.

\subsection{Задача 4.3. Збіжність у метриці Хаусдорфа}
Нехай $\{\mathcal{A}_k\}\subset\mathrm{comp}(\mathbb{R}^n)$~--- послідовність компактів,
$\mathcal{A}_k\supset\mathcal{A}_{k+1}$, $k=1,2,\ldots$,
$\mathcal{A}=\bigcap_{k=1}^{\infty}\mathcal{A}_k$. Довести, що
$\mathcal{A}\in\mathrm{comp}(\mathbb{R}^n)$ і в метриці Хаусдорфа
\begin{equation*}
  \lim_{k\to\infty}\mathcal{A}_k=\mathcal{A}.
\end{equation*}

\begin{korotko}
Перетин спадної послідовності компактів~--- компакт (теорема Кантора).
У метриці Хаусдорфа достатньо оцінити $\sup_{y\in\mathcal{A}_k}d(y,\mathcal{A})$;
від супротивного~--- точки на відстані $\ge\varepsilon$ від $\mathcal{A}$ дають
збіжну підпослідовність у $\mathcal{A}_1$, границя лежить у кожному $\mathcal{A}_N$,
отже в $\mathcal{A}$.
\end{korotko}

\noindent\textbf{Доведення.}

\noindent\textbf{Частина~1: компактність $\mathcal{A}$.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item \textbf{Замкненість.} Кожен $\mathcal{A}_k$ компактний, отже замкнений.
        Перетин довільної кількості замкнених множин замкнений, тому
        $\mathcal{A}=\bigcap_{k=1}^{\infty}\mathcal{A}_k$ замкнена.
  \item \textbf{Обмеженість.} $\mathcal{A}\subset\mathcal{A}_k$ для всіх $k\ge1$;
        зокрема $\mathcal{A}\subset\mathcal{A}_1$, а $\mathcal{A}_1$ обмежений.
  \item \textbf{Непорожність.} За теоремою Кантора про вкладені компакти перетин
        спадної послідовності непорожніх компактів непорожній (і компактний).
\end{enumerate}
Отже $\mathcal{A}$ замкнена, обмежена і непорожня в $\mathbb{R}^n$, тобто
$\mathcal{A}\in\mathrm{comp}(\mathbb{R}^n)$.

\noindent\textbf{Частина~2: збіжність у метриці Хаусдорфа.}
Нагадаємо метрику Хаусдорфа:
\begin{equation*}
  h(\mathcal{A}_k,\mathcal{A})
  =\max\Bigl\{\sup_{y\in\mathcal{A}_k}d(y,\mathcal{A}),\;
  \sup_{x\in\mathcal{A}}d(x,\mathcal{A}_k)\Bigr\}.
\end{equation*}
Оскільки $\mathcal{A}\subset\mathcal{A}_k$, то $\sup_{x\in\mathcal{A}}d(x,\mathcal{A}_k)=0$, і
\begin{equation*}
  h(\mathcal{A}_k,\mathcal{A})=\sup_{y\in\mathcal{A}_k}d(y,\mathcal{A}).
\end{equation*}

Доведемо від супротивного. Якщо $\lim_{k\to\infty}h(\mathcal{A}_k,\mathcal{A})\neq0$,
то існують $\varepsilon>0$ та підпослідовність $k_m\to\infty$ такі, що
\begin{equation*}
  \sup_{y\in\mathcal{A}_{k_m}}d(y,\mathcal{A})\ge\varepsilon.
\end{equation*}
Для кожного $m$ знайдеться $y_m\in\mathcal{A}_{k_m}$ з $d(y_m,\mathcal{A})\ge\varepsilon$.

Уся послідовність $\{y_m\}$ лежить у компакті $\mathcal{A}_1$; виділимо збіжну
підпослідовність $y_{m_j}\to y^*\in\mathcal{A}_1$. З неперервності відстані
\begin{equation*}
  d(y^*,\mathcal{A})=\lim_{j\to\infty}d(y_{m_j},\mathcal{A})\ge\varepsilon>0,
\end{equation*}
тобто $y^*\notin\mathcal{A}$.

Зафіксуємо довільне $N\in\mathbb{N}$. Для достатньо великих $j$ маємо $k_{m_j}\ge N$,
отже через вкладеність $y_{m_j}\in\mathcal{A}_{k_{m_j}}\subset\mathcal{A}_N$.
Оскільки $\mathcal{A}_N$ замкнена, $y^*\in\mathcal{A}_N$. Це для всіх $N$, тому
$y^*\in\bigcap_{N=1}^{\infty}\mathcal{A}_N=\mathcal{A}$~--- суперечність.

Отже $\lim_{k\to\infty}h(\mathcal{A}_k,\mathcal{A})=0$, тобто
$\lim_{k\to\infty}\mathcal{A}_k=\mathcal{A}$ у метриці Хаусдорфа.

\subsection{Задача 4.4. Відстань Хаусдорфа для множини Кантора}
Нехай $\mathcal{K}$~--- множина Кантора, $\mathcal{K}_0=[0,1]$,
\begin{gather*}
  \mathcal{K}_1=\Bigl[0,\tfrac13\Bigr]\cup\Bigl[\tfrac23,1\Bigr],\\
  \mathcal{K}_2=\Bigl[0,\tfrac{1}{3^2}\Bigr]\cup\Bigl[\tfrac{2}{3^2},\tfrac13\Bigr]
  \cup\Bigl[\tfrac23,\tfrac23+\tfrac{1}{3^2}\Bigr]
  \cup\Bigl[\tfrac23+\tfrac{2}{3^2},1\Bigr],\quad\ldots
\end{gather*}
Знайти відстань Хаусдорфа
$h(\mathcal{K},\mathcal{K}_0)$, $h(\mathcal{K},\mathcal{K}_1)$,
$h(\mathcal{K},\mathcal{K}_2)$, $\ldots$

\begin{korotko}
Оскільки $\mathcal{K}\subset\mathcal{K}_n$, маємо
$h(\mathcal{K},\mathcal{K}_n)=\sup_{y\in\mathcal{K}_n}d(y,\mathcal{K})$.
Найвіддаленіша точка~--- середина «дірки», що вилучається на $(n{+}1)$-му кроці;
довжина дірки $1/3^{n+1}$, тому $h(\mathcal{K},\mathcal{K}_n)=1/(2\cdot3^{n+1})$.
\end{korotko}

\noindent\textbf{Доведення.}
За означенням метрики Хаусдорфа
\begin{equation*}
  h(\mathcal{K},\mathcal{K}_n)
  =\max\Bigl\{\sup_{x\in\mathcal{K}}d(x,\mathcal{K}_n),\;
  \sup_{y\in\mathcal{K}_n}d(y,\mathcal{K})\Bigr\}.
\end{equation*}
Оскільки $\mathcal{K}\subset\mathcal{K}_n$, то $\sup_{x\in\mathcal{K}}d(x,\mathcal{K}_n)=0$, і
\begin{equation*}
  h(\mathcal{K},\mathcal{K}_n)=\sup_{y\in\mathcal{K}_n}d(y,\mathcal{K}).
\end{equation*}
Потрібно знайти точку $y\in\mathcal{K}_n$, найвіддаленішу від $\mathcal{K}$.

Множина $\mathcal{K}_n$ складається з $2^n$ замкнених відрізків довжини $1/3^n$.
Візьмемо довільний такий відрізок $I\in\mathcal{K}_n$. Найвіддаленіша від $\mathcal{K}$
точка лежить посередині найбільшої «дірки» (вилученого інтервалу) всередині $I$.
Найбільша дірка з'являється на $(n{+}1)$-му кроці побудови; її довжина
\begin{equation*}
  L_{n+1}=\frac13\cdot\frac{1}{3^n}=\frac{1}{3^{n+1}}.
\end{equation*}
Найвіддаленіша точка $y^*$~--- середина цього інтервалу; відстань від $y^*$ до
кінців дірки (які належать $\mathcal{K}$) дорівнює половині довжини:
\begin{equation*}
  \sup_{y\in\mathcal{K}_n}d(y,\mathcal{K})=\frac{L_{n+1}}{2}=\frac{1}{2\cdot3^{n+1}}.
\end{equation*}
Отже
\begin{equation*}
  h(\mathcal{K},\mathcal{K}_n)=\frac{1}{2\cdot3^{n+1}}.
\end{equation*}

Конкретні значення:
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item $n=0$:\quad $h(\mathcal{K},\mathcal{K}_0)=\dfrac{1}{2\cdot3}=\dfrac16$
        (середина $(1/3,2/3)$~--- точка $1/2$, відстань до кінців $1/6$);
  \item $n=1$:\quad $h(\mathcal{K},\mathcal{K}_1)=\dfrac{1}{2\cdot3^2}=\dfrac{1}{18}$;
  \item $n=2$:\quad $h(\mathcal{K},\mathcal{K}_2)=\dfrac{1}{2\cdot3^3}=\dfrac{1}{54}$;
  \item загалом $h(\mathcal{K},\mathcal{K}_n)=\dfrac{1}{2\cdot3^{n+1}}$~---
        геометрична прогресія з кожним кроком побудови.
\end{enumerate}

\paragraph{Умови для задач 4.7--4.11.}
У $\mathbb{R}^2$ задано одиничний квадрат $\mathcal{K}$ з вершинами в точках
$a^{(1)},a^{(2)},a^{(3)},a^{(4)}$. Позначимо через $F$ систему перетворень
\begin{equation*}
  f_i(x)=r(x-a^{(i)})+a^{(i)},\quad i=1,2,3,4,
\end{equation*}
де $r\in(0,\tfrac12)$, $x\in\mathbb{R}^2$.
Геометрично $f_i$~--- гомотетія (стиск) з коефіцієнтом $r$ і центром у вершині
$a^{(i)}$; послідовність $\mathcal{K}_{j+1}=\bigcup_{i=1}^4 f_i(\mathcal{K}_j)$.

\subsection{Задача 4.7. Перші ітерації IFS на квадраті}
Побудувати геометричне представлення перших трьох членів послідовності
$\mathcal{K}_j$, $j=1,2,3$, де $\mathcal{K}_1=\mathcal{K}$,
$\mathcal{K}_{j+1}=f(\mathcal{K}_j)$, $j=1,2,3$,
\begin{equation*}
  f=f_1\cup f_2\cup f_3\cup f_4,\quad
  a^{(1)}=\begin{pmatrix}0\\0\end{pmatrix},\;
  a^{(2)}=\begin{pmatrix}1\\0\end{pmatrix},\;
  a^{(3)}=\begin{pmatrix}0\\1\end{pmatrix},\;
  a^{(4)}=\begin{pmatrix}1\\1\end{pmatrix}.
\end{equation*}

\begin{korotko}
$\mathcal{K}_1$~--- одиничний квадрат; $\mathcal{K}_2$~--- чотири квадрати зі
стороною $r$ у кутах; $\mathcal{K}_3$~--- $16$ квадратів зі стороною $r^2$
(по $4$ у кожному куті попередніх).
\end{korotko}

\begin{enumerate}[leftmargin=1.6em,itemsep=0.35em]
  \item \textbf{$\mathcal{K}_1$.} Початкова множина $\mathcal{K}=[0,1]\times[0,1]$.
  \item \textbf{$\mathcal{K}_2$.} Застосовуємо всі чотири гомотетії до $\mathcal{K}_1$.
        Отримуємо чотири квадрати зі стороною $r$, «притиснуті» до кутів $a^{(i)}$.
        Оскільки $r<1/2$, квадрати не перетинаються (між ними хрестоподібна порожнеча):
        \[
          [0,r]\times[0,r],\quad [1-r,1]\times[0,r],\quad
          [0,r]\times[1-r,1],\quad [1-r,1]\times[1-r,1].
        \]
  \item \textbf{$\mathcal{K}_3$.} Кожна $f_i$ застосовується до чотирьох квадратів з
        $\mathcal{K}_2$; всього $4\times4=16$ квадратів зі стороною $r^2$, згрупованих
        по $4$ у кутах кожного квадрата попереднього кроку.
\end{enumerate}

\subsection{Задача 4.8. Алгоритм атрактора IFS}
Розробити алгоритм знаходження атрактора системи функцій $F$.

\begin{korotko}
Детерміновано: $S_{j+1}=\bigcup_{i=1}^4 f_i(S_j)$, атрактор~---
$\mathcal{A}=\lim_{j\to\infty}S_j$ (або $\bigcap_j\mathcal{K}_j$ при $S_0=\mathcal{K}$).
Стохастично: chaos game~--- випадковий вибір $f_i$ і ітерація $x_{j+1}=f_i(x_j)$.
\end{korotko}

\textbf{1. Детермінований алгоритм (Хатчінсон).}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item \textbf{Крок~0:} обрати непорожню компактну $S_0$ (зручно $S_0=\mathcal{K}$).
  \item \textbf{Крок~$j$:} $S_{j+1}=\bigcup_{i=1}^4 f_i(S_j)$.
  \item \textbf{Границя:} $\mathcal{A}=\lim_{j\to\infty}S_j$ у метриці Хаусдорфа;
        якщо $S_0=\mathcal{K}$, то $\mathcal{A}=\bigcap_{j=1}^{\infty}\mathcal{K}_j$.
\end{enumerate}

\textbf{2. Стохастичний алгоритм (гра в хаос).}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item \textbf{Крок~0:} обрати $x_0\in\mathbb{R}^2$.
  \item \textbf{Крок~$j$:} випадково (з імовірністю $1/4$) обрати $f_i$ і покласти
        $x_{j+1}=f_i(x_j)$.
  \item \textbf{Результат:} відкинути перші $N$ точок (наприклад, $N=100$); решта
        $\{x_k\}_{k=N}^{\infty}$ щільно заповнює $\mathcal{A}$.
\end{enumerate}

\subsection{Задача 4.9. Фрактальна розмірність атрактора}
Знайти фрактальну розмірність атрактора, який відповідає системі функцій $F$.

\begin{korotko}
На кроці $j$ маємо $N_j=4^{j-1}$ квадратів зі стороною $\varepsilon_j=r^{j-1}$;
$D_f=\ln N_j/\ln(1/\varepsilon_j)=\ln4/\ln(1/r)$.
\end{korotko}

На кроці $j$ множина $\mathcal{K}_j$ складається з $N_j=4^{j-1}$ квадратів зі
стороною (діаметром, з точністю до $\sqrt2$) $\varepsilon_j=r^{j-1}$; цих множин
достатньо для $\varepsilon_j$-покриття атрактора $\mathcal{A}$.

За означенням ємнісної (фрактальної) розмірності
\begin{equation*}
  D_f(\mathcal{A})
  =\lim_{j\to\infty}\frac{\ln N_j}{\ln(1/\varepsilon_j)}
  =\lim_{j\to\infty}\frac{\ln(4^{j-1})}{\ln(1/r^{j-1})}
  =\lim_{j\to\infty}\frac{(j-1)\ln4}{(j-1)\ln(1/r)}
  =\frac{\ln4}{\ln(1/r)}=-\frac{\ln4}{\ln r}.
\end{equation*}

\subsection{Задача 4.10. Розмірність подібності}
Знайти розмірність подібності атрактора системи функцій $F$. Порівняти її
з фрактальною розмірністю.

\begin{korotko}
Рівняння Морана: $4r^{D_s}=1$, звідки $D_s=\ln4/\ln(1/r)=D_f$ (за теоремою
Хатчінсона--Морана при умові відкритих множин).
\end{korotko}

Система складається з чотирьох подібних перетворень з коефіцієнтом стиску $r_i=r$.
Розмірність подібності $D_s$~--- розв'язок рівняння Морана
\begin{equation*}
  \sum_{i=1}^{4}r_i^{D_s}=1 \quad\Longrightarrow\quad 4r^{D_s}=1,\quad
  r^{D_s}=\tfrac14.
\end{equation*}
Логарифмуємо:
\begin{equation*}
  D_s\ln r=\ln(1/4)=-\ln4 \quad\Longrightarrow\quad
  D_s=-\frac{\ln4}{\ln r}=\frac{\ln4}{\ln(1/r)}.
\end{equation*}

\textbf{Порівняння.}
Для $r<1/2$ виконується \emph{умова відкритих множин} (OSC): існує відкрита
$V=(0,1)\times(0,1)$ така, що $f_i(V)\subset V$ і $f_i(V)\cap f_j(V)=\emptyset$
при $i\neq j$. За теоремою Хатчінсона--Морана при OSC розмірність Хаусдорфа
(і ємнісна фрактальна розмірність) збігається з розмірністю подібності:
\begin{equation*}
  D_f(\mathcal{A})=D_s(\mathcal{A})=\frac{\ln4}{\ln(1/r)}.
\end{equation*}

\subsection{Задача 4.11. Умова відкритої множини}
Перевірити для системи функцій $F$ виконання умов відкритої множини.
Обґрунтувати самоподібність атрактора системи функцій $F$.

\begin{korotko}
$V=(0,1)\times(0,1)$: образи $f_i(V)$~--- квадрати зі стороною $r$ у кутах;
при $r<1/2$ вони не перетинаються і лежать у $V$. Атрактор
$\mathcal{A}=\bigcup_{i=1}^4 f_i(\mathcal{A})$~--- 4 неперетинні копії, зменшені в $1/r$ разів.
\end{korotko}

\textbf{1. Умова відкритої множини (OSC).}
Потрібна непорожня відкрита $V\subset\mathbb{R}^2$ така, що $f_i(V)\subset V$ і
$f_i(V)\cap f_j(V)=\emptyset$ при $i\neq j$.

Оберемо $V=(0,1)\times(0,1)$. Тоді $f_i(V)$~--- відкритий квадрат зі стороною $r$,
«притиснутий» до вершини $a^{(i)}$; координати лежать у $[0,r]$ та $[1-r,1]$.
Оскільки $r<1/2$, маємо $2r<1$ і $r<1-r$, тому проекції квадратів на осі не
перетинаються в центрі, отже $f_i(V)\cap f_j(V)=\emptyset$ при $i\neq j$.
Також $r<1$, тому $f_i(V)\subset V$. OSC виконана.

\textbf{2. Самоподібність атрактора.}
Атрактор $\mathcal{A}$ задовольнює рівняння Хатчінсона
\begin{equation*}
  \mathcal{A}=\bigcup_{i=1}^{4}f_i(\mathcal{A}).
\end{equation*}
Кожне $f_i$~--- гомотетія з коефіцієнтом $r$; при $r<1/2$ копії $f_i(\mathcal{A})$
не перетинаються (перетин можливий лише по границі нульової міри). Тому $\mathcal{A}$
складається з чотирьох точних власних копій, зменшених у $1/r$ разів,~--- строго
самоподібний фрактал.

\subsection{Задача 4.12. Килим Серпінського}
Розробити алгоритм та побудувати зображення килима Серпінського.

\begin{korotko}
Кожен квадрат ділиться на $3\times3$; центральний прибирають. Границя:
$\mathcal{S}=\bigcap_{k=0}^{\infty}\mathcal{S}_k$.
\end{korotko}

\textbf{Алгоритм (детермінований).}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item \textbf{Крок~0:} $\mathcal{S}_0$~--- суцільний одиничний квадрат.
  \item \textbf{Крок~$k$:} кожен квадрат з $\mathcal{S}_{k-1}$ ділиться на $9$ рівних
        частин ($3\times3$); центральний відкритий квадрат видаляється (залишається $8$).
  \item \textbf{Границя:} килим Серпінського $\mathcal{S}=\bigcap_{k=0}^{\infty}\mathcal{S}_k$.
\end{enumerate}

Нехай $f\colon I\to I$~--- неперервна функція на інтервалі $I\subset\mathbb{R}$.

\subsection{Задача 5.1. Період $n\geqslant 3$ і нерухома точка}
Довести, що якщо функція $f$ має орбіту періоду $n\geqslant 3$, то вона має
орбіту періоду~$1$ (тобто має нерухому точку).

\begin{korotko}
На орбіті $\{x_1,\ldots,x_n\}$ беремо $a=\min$, $b=\max$; тоді $f(a)\ge a$,
$f(b)\le b$. Для $g(x)=f(x)-x$ на $[a,b]$ за Больцано--Коші існує $c$ з $f(c)=c$.
\end{korotko}

\noindent\textbf{Доведення.}
Нехай $O(x_1)=\{x_1,x_2,\ldots,x_n\}$~--- орбіта періоду $n$:
$f(x_i)=x_{i+1}$ при $i<n$, $f(x_n)=x_1$, усі точки різні.
Оскільки орбіта скінченна, існують
\begin{equation*}
  a=\min\{x_1,\ldots,x_n\},\qquad b=\max\{x_1,\ldots,x_n\}.
\end{equation*}
Оскільки $a$ належить орбіті, $f(a)$ теж на орбіті, але $a$~--- найменший, тому
$f(a)\ge a$, тобто $f(a)-a\ge0$. Аналогічно $f(b)\le b$, тобто $f(b)-b\le0$.

Для $g(x)=f(x)-x$ на $[a,b]$ маємо $g(a)\ge0$ і $g(b)\le0$. За теоремою
Больцано--Коші існує $c\in[a,b]$ з $g(c)=0$, отже $f(c)=c$~--- нерухома точка
(орбіта періоду~$1$).

\textit{Зауваження.} Доказ працює для будь-якого $n\ge2$; умова $n\ge3$
підкреслює контекст теореми Шарковського.

\subsection{Задача 5.2*. Періоди $2^n$ та $2^k$}
Довести, що якщо $f$ має орбіту періоду $2^n$, то вона має орбіту періоду
$2^k$ при $n>k$.

\begin{korotko}
З теореми Шарковського: $2^n\rhd 2^k$ при $n>k$. Альтернатива~---
$g=f^{2^{n-1}}$ має період~$2$, отже нерухому точку; її період для $f$
ділить $2^{n-1}$.
\end{korotko}

\noindent\textbf{Доведення.}
Це наслідок \textbf{теореми Шарковського}. Лінійний порядок $\rhd$:
\begin{equation*}
  3\rhd5\rhd7\rhd\cdots\rhd2\cdot3\rhd2\cdot5\rhd\cdots\rhd
  2^m\cdot3\rhd\cdots\rhd2^n\rhd2^{n-1}\rhd\cdots\rhd2\rhd1.
\end{equation*}
Якщо неперервна $f\colon I\to I$ має періодичну точку періоду $m$ і $m\rhd l$,
то існує періодична точка періоду $l$. Оскільки $2^n\rhd2^k$ при $n>k$,
твердження випливає безпосередньо.

\medskip
\noindent\textbf{Конструктивне доведення (без повної теореми).}
Достатньо показати: період $2^n$ ($n\ge1$) $\Rightarrow$ період $2^{n-1}$;
далі індукція дає всі $2^k$ при $k<n$.

Нехай $f$ має орбіту періоду $2^n$. Покладемо $g(x)=f^{2^{n-1}}(x)$.
Для $g$ початкова орбіта розбивається на орбіти періоду~$2$, бо
$g(g(x))=f^{2^n}(x)=x$ і $g(x)\neq x$ (інакше період $f$ ділив би $2^{n-1}$).

Отже $g$ має орбіту періоду~$2$; за задачею~5.1 (для $n\ge2$) $g$ має нерухому
точку $y$: $g(y)=y$, тобто $f^{2^{n-1}}(y)=y$. Точка $y$ періодична для $f$,
її період ділить $2^{n-1}$.

Щоб отримати точку \emph{найменшого} періоду саме $2^{n-1}$ (а не меншого),
в теорії динаміки використовують леми про перетин графіків ітерацій з
діагоналлю $y=x$: період $2^n$ забезпечує достатню кількість таких перетинів.
Крок за кроком спускаючись, отримуємо орбіти всіх періодів $2^k$ при $k<n$.

\paragraph{Умови для задач 5.3--5.5.}
Нехай $Q(x)=4x(1-x)$, $0\leqslant x\leqslant 1$, та
\begin{equation*}
  T(x)=
  \begin{cases}
    2x, & 0\leqslant x\leqslant \tfrac12,\\[0.3em]
    2(1-x), & \tfrac12<x\leqslant 1.
  \end{cases}
\end{equation*}

\subsection{Задача 5.3*. Топологічна спряженість $Q$ і $T$}
Показати, що відображення $Q$ і $T$ топологічно спряжені на $[0,1]$ за
допомогою $H(x)=\sin^2\!\bigl(\tfrac{\pi}{2}x\bigr)$, тобто
$Q=H\circ T\circ H^{-1}$.

\begin{korotko}
$H$~--- гомеоморфізм $[0,1]\to[0,1]$ (строго зростає). Перевіряємо
$Q(H(x))=H(T(x))=\sin^2(\pi x)$ для обох гілок $T$.
\end{korotko}

\noindent\textbf{Доведення.}

\textbf{1. $H$~--- гомеоморфізм на $[0,1]$.}
$H(x)=\sin^2(\tfrac{\pi}{2}x)$ неперервна як композиція неперервних функцій;
$H(0)=0$, $H(1)=1$. На $(0,1)$
\[
  H'(x)=\frac{\pi}{2}\sin(\pi x)>0,
\]
отже $H$ строго зростає на $[0,1]$, отже бієктивна $[0,1]\to[0,1]$.
Обернена $H^{-1}(x)=\tfrac{2}{\pi}\arcsin(\sqrt{x})$ неперервна, тому $H$~---
гомеоморфізм.

\textbf{2. Умова спряження: $Q(H(x))=H(T(x))$.}
Якщо $0\le x\le\tfrac12$, то $T(x)=2x$ і
\[
  H(T(x))=\sin^2(\pi x).
\]
Якщо $\tfrac12<x\le1$, то $T(x)=2(1-x)$ і
\[
  H(T(x))=\sin^2(\pi-\pi x)=\sin^2(\pi x).
\]
Тому для всіх $x\in[0,1]$ маємо $H(T(x))=\sin^2(\pi x)$.

Далі
\begin{align*}
  Q(H(x)) &= 4\sin^2\!\Bigl(\tfrac{\pi}{2}x\Bigr)
            \Bigl(1-\sin^2\!\Bigl(\tfrac{\pi}{2}x\Bigr)\Bigr)
           = 4\sin^2\!\Bigl(\tfrac{\pi}{2}x\Bigr)\cos^2\!\Bigl(\tfrac{\pi}{2}x\Bigr) \\
          &= \Bigl(2\sin\!\Bigl(\tfrac{\pi}{2}x\Bigr)\cos\!\Bigl(\tfrac{\pi}{2}x\Bigr)\Bigr)^2
           = \sin^2(\pi x).
\end{align*}
Отже $Q(H(x))=H(T(x))$, тобто $Q=H\circ T\circ H^{-1}$: відображення
топологічно спряжені.

\subsection{Задача 5.4. Орбіта періоду 3}
Знайти орбіту $Q(x)$ періоду~$3$, показавши, що
$\bigl\{\tfrac27,\tfrac47,\tfrac67\bigr\}$~--- 3-цикл для $T(x)$.

\begin{korotko}
$T(2/7)=4/7$, $T(4/7)=6/7$, $T(6/7)=2/7$. Через $H$ отримуємо 3-цикл для $Q$:
$\{\sin^2(\pi/7),\,\sin^2(2\pi/7),\,\sin^2(3\pi/7)\}$.
\end{korotko}

\noindent\textbf{Доведення.}
Перевіримо 3-цикл для $T$:
\begin{align*}
  T(2/7) &= 2\cdot(2/7)=4/7 \quad (\le 1/2),\\
  T(4/7) &= 2\bigl(1-4/7\bigr)=6/7 \quad (> 1/2),\\
  T(6/7) &= 2\bigl(1-6/7\bigr)=2/7.
\end{align*}
Орбіта замкнулась на 3-му кроці, тому $\{2/7,4/7,6/7\}$~--- 3-цикл для $T$.

Оскільки $Q$ і $T$ спряжені через $H$, періодичні орбіти $T$ переходять у
орбіти $Q$ того ж періоду: $y_i=H(x_i)=\sin^2(\tfrac{\pi}{2}x_i)$.
\begin{align*}
  y_1 &= \sin^2\!\Bigl(\tfrac{\pi}{7}\Bigr), &
  y_2 &= \sin^2\!\Bigl(\tfrac{2\pi}{7}\Bigr), &
  y_3 &= \sin^2\!\Bigl(\tfrac{3\pi}{7}\Bigr).
\end{align*}
Шуканий 3-цикл для $Q(x)$:
\[
  \Bigl\{\sin^2\!\Bigl(\tfrac{\pi}{7}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{2\pi}{7}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{3\pi}{7}\Bigr)\Bigr\}.
\]

\subsection{Задача 5.5. Орбіти періодів 4 і 5}
Знайти орбіти $Q(x)$ періодів~4 і~5.

\begin{korotko}
Цикли $T$ будують з раціональних точок зі знаменником $2^n\pm1$; для $Q$
застосовуємо $H(x)=\sin^2(\tfrac{\pi}{2}x)$.
\end{korotko}

\noindent\textbf{Доведення.}
Використовуємо спряження: знаходимо цикл для $T$, потім переводимо через $H$.

\textbf{1. Період~4.}
Для $n=4$ розглянемо знаменник $2^4-1=15$. Перевірка $x_1=2/15$ дає
$T(14/15)=4/15$ (передперіодичність). Для $x_1=1/15$:
\[
  \tfrac{1}{15}\xrightarrow{T}\tfrac{2}{15}\xrightarrow{T}\tfrac{4}{15}
  \xrightarrow{T}\tfrac{8}{15}\xrightarrow{T}\tfrac{14}{15}\xrightarrow{T}\tfrac{2}{15}.
\]
4-цикл для $T$: $\{2/15,\,4/15,\,8/15,\,14/15\}$.
Відповідна орбіта для $Q$:
\[
  \Bigl\{\sin^2\!\Bigl(\tfrac{\pi}{15}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{2\pi}{15}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{4\pi}{15}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{7\pi}{15}\Bigr)\Bigr\}.
\]

\textbf{2. Період~5.}
Для $n=5$ знаменник $2^5-1=31$. Починаємо з $x_1=2/31$:
\[
  \tfrac{2}{31}\xrightarrow{T}\tfrac{4}{31}\xrightarrow{T}\tfrac{8}{31}
  \xrightarrow{T}\tfrac{16}{31}\xrightarrow{T}\tfrac{30}{31}\xrightarrow{T}\tfrac{2}{31}.
\]
5-цикл для $T$: $\{2/31,\,4/31,\,8/31,\,16/31,\,30/31\}$.
Відповідна орбіта для $Q$:
\[
  \Bigl\{\sin^2\!\Bigl(\tfrac{\pi}{31}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{2\pi}{31}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{4\pi}{31}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{8\pi}{31}\Bigr),\;
        \sin^2\!\Bigl(\tfrac{15\pi}{31}\Bigr)\Bigr\}.
\]

\subsection{Задача 5.7. Орбіта в $\mathbb{C}$}
Для многочлена $f(z)=z^2-1+0.213i$ реалізувати алгоритм знаходження орбіти
на 100 ітерацій при $z_0=0.3+0.2i$ та $z_0=0.31+0.2i$.

\begin{korotko}
Ітерації $z_{k+1}=z_k^2+c$, $c=-1+0.213i$, $k=0,\ldots,99$. Точки $z_0$ відрізняються
на $0.01$ у $\Re z$~--- орбіти розходяться (чутливість до початкових умов).
\end{korotko}

Розглянемо квадратичне сімейство $f_c(z)=z^2+c$, $c\in\mathbb{C}$.

\textbf{Алгоритм.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Задати $z_0$ і $c=-1+0.213i$.
  \item Обчислити $z_{k+1}=z_k^2+c$ для $k=0,1,\ldots,99$.
  \item Зберегти всі $z_k$ для аналізу та візуалізації на комплексній площині.
\end{enumerate}

Початкові точки $z_0=0.3+0.2i$ та $z_0=0.31+0.2i$ відрізняються незначно,
але через нелінійність $z\mapsto z^2$ орбіти з часом суттєво розходяться
(рис.~\ref{fig:julia-57-510}, лівий верхній графік).

\subsection{Задача 5.8. Множина Жюліa ($c=-0.8+0.156i$)}
Реалізувати алгоритм конструювання множини Жюліa для
$f(z)=z^2-0.8+0.156i$.

\begin{korotko}
Алгоритм часу втечі: $|z_k|>2$ $\Rightarrow$ орбіта тікає; інакше точка в
$K(f_c)$. Для $c=-0.8+0.156i$~--- зв'язна «драконоподібна» множина Жюліa.
\end{korotko}

\subsection{Задача 5.9. Множина Жюліa ($c=0.285+0.01i$)}
Реалізувати алгоритм конструювання множини Жюліa для
$f(z)=z^2+0.285+0.01i$.

\begin{korotko}
Той самий escape-time алгоритм. Для $c=0.285+0.01i$~--- «пухнасті» компоненти,
майже роз'єднані (Fatou dust).
\end{korotko}

\subsection{Задача 5.10. Множина Жюліa ($c=-0.0085+0.71i$)}
Реалізувати алгоритм конструювання множини Жюліa для
$f(z)=z^2-0.0085+0.71i$.

\begin{korotko}
Для $c=-0.0085+0.71i$~--- тонка дендритна структура з витонченими гілками.
\end{korotko}

\paragraph{Алгоритм escape-time (задачі 5.8--5.10).}
Множина Жюліa $J(f_c)$~--- межа точок з обмеженими орбітами; заповнена множина
$K(f_c)$ містить усі такі точки. Якщо для деякого $k$ виконується
$|z_k|>\max(|c|,2)$ (на практиці часто беруть поріг $R=2$), орбіта гарантовано
прямує до $\infty$.

\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Обрати прямокутник у $\mathbb{C}$, наприклад
        $[-1.5,1.5]\times[-1.5,1.5]$.
  \item Розбити на сітку пікселів (наприклад, $800\times800$); кожен піксель~---
        початкова точка $z_0$.
  \item Задати $N_{\max}$ (100 або 300 ітерацій).
  \item Для кожного $z_0$: обчислювати $z_{k+1}=z_k^2+c$; якщо $|z_k|>2$ на кроці
        $k$, зафарбувати піксель кольором, що залежить від $k$; якщо після
        $N_{\max}$ ітерацій $|z_k|\le2$, зафарбувати в чорний ($K(f_c)$).
\end{enumerate}

\subsection{Задача 5.11. Множина Мандельброта}
Розробити алгоритм конструювання множини Мандельброта.

\begin{korotko}
$M=\{c\in\mathbb{C}:\text{орбіта }0\text{ при }z_{n+1}=z_n^2+c\text{ обмежена}\}$.
Escape-time: для кожного $c$ ітеруємо з $z=0$; $|z|>2$ $\Rightarrow$ $c\notin M$.
\end{korotko}

Множина Мандельброта $M$~--- множина параметрів $c$, для яких орбіта нуля під
$f_c(z)=z^2+c$ залишається обмеженою ($z_0=0$, $z_{n+1}=z_n^2+c$).

\textbf{Алгоритм escape-time.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Область: $c=x+iy$, наприклад $x\in[-2.5,1.0]$, $y\in[-1.5,1.5]$.
  \item Сітка пікселів; кожен піксель~--- значення $c$.
  \item Для кожного $c$: $z\leftarrow0$, ітерації $z\leftarrow z^2+c$ (до $N_{\max}$).
  \item Якщо $|z|>2$ на кроці $k<N_{\max}$, то $c\notin M$; колір залежить від $k$.
  \item Якщо після $N_{\max}$ ітерацій $|z|\le2$, вважаємо $c\in M$ (чорний колір).
\end{enumerate}

Приклад побудови~--- рис.~\ref{fig:mandelbrot-511-513} (ліворуч).

\subsection{Задача 5.12. Крива «готель Гільберта»}
Розробити алгоритм та побудувати зображення кривої «готель Гільберта».

\begin{korotko}
Рекурсія: квадрат ділиться на $2\times2$; чотири копії $H_{n-1}$ з поворотами
і дзеркаленнями з'єднуються відрізками. На практиці малюють $H_k$, $k=5$--$6$.
\end{korotko}

Крива Гільберта~--- неперервна space-filling крива в одиничному квадраті.

\textbf{Алгоритм (рекурсивний поділ).}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item \textbf{$H_1$:} квадрат ділиться на $2\times2$; центри чотирьох підквадратів
        з'єднують трьома відрізками (П-подібна крива).
  \item \textbf{$H_n$:} квадрат розбивається на $4^n$ підквадратів. З чотирьох копій
        $H_{n-1}$ (кожна зменшена вдвічі):
        \begin{itemize}[leftmargin=1.2em,itemsep=0.15em]
          \item ліва нижня~--- поворот $90^\circ$ за годинниковою + дзеркалення;
          \item ліва верхня і права верхня~--- лише зменшення і перенесення;
          \item права нижня~--- поворот $90^\circ$ проти годинникової + дзеркалення.
        \end{itemize}
        Кінці копій з'єднують трьома додатковими відрізками.
  \item \textbf{Границя:} $H=\lim_{n\to\infty}H_n$; на практиці будують $H_k$
        ($k=5$ на рис.~\ref{fig:mandelbrot-511-513}, центр).
\end{enumerate}

\subsection{Задача 5.14. Множина Мандельброта для $z^3+c$}
Отримати зображення множини Мандельброта для $f(z)=z^3+c$; показати, що якщо
$|c|>2$, орбіта прямує до $\infty$.

\begin{korotko}
При $|c|>2$: $z_1=c$, далі $|z_{n+1}|\ge|z_n|(|z_n|^2-1)>3|z_n|$, отже
$|z_n|>3^{n-1}|c|\to\infty$. Multibrot ($d=3$) будують escape-time з $z_0=0$.
\end{korotko}

\noindent\textbf{Доведення:} $|c|>2$ $\Rightarrow$ орбіта $z_0=0$ тікає.
\begin{align*}
  z_0 &= 0, & z_1 &= c, & |z_1| &= |c|>2.
\end{align*}
Індукція: якщо $|z_n|\ge|c|>2$, то
\begin{equation*}
  |z_{n+1}|=|z_n^3+c|\ge|z_n|^3-|c|\ge|z_n|^3-|z_n|
  =|z_n|(|z_n|^2-1).
\end{equation*}
Оскільки $|z_n|>2$, маємо $|z_n|^2-1>3$, тому $|z_{n+1}|>3|z_n|$ і
$|z_n|>3^{n-1}|c|\to\infty$ при $n\to\infty$.

Multibrot степеня~$3$ будують аналогічно до класичної множині Мандельброта:
для кожного $c$ ітеруємо $z_{n+1}=z_n^3+c$ з $z_0=0$; якщо $|z|>2$ не настає
за $N_{\max}$ ітерацій, $c$ вважають належним множині (рис.~\ref{fig:multibrot-514-516}, ліворуч).

\subsection{Задача 5.15. Басейн $\mathcal{A}(1)$ для $z^4-1$}
Побудувати графічне зображення басейну притягання $\mathcal{A}(1)$ для
$f(z)=z^4-1$.

\begin{korotko}
Басейн Ньютона для кореня $z=1$: $N(z)=z-f/f'$, тут
$z_{k+1}=\dfrac{3z_k^4+1}{4z_k^3}$. Зафарбовуємо точки, що збігаються до $1$.
\end{korotko}

Область притягання кореня $z=1$ для $f(z)=z^4-1$~--- \textbf{басейн Ньютона}
методу $N(z)=z-f(z)/f'(z)$. Оскільки $f'(z)=4z^3$,
\begin{equation*}
  z_{k+1}=z_k-\frac{z_k^4-1}{4z_k^3}=\frac{3z_k^4+1}{4z_k^3}.
\end{equation*}
$\mathcal{A}(1)$~--- фрактальна множина точок, орбіти яких під $N$ збігаються до
$1$ (рис.~\ref{fig:multibrot-514-516}, центр).

\subsection{Задача 5.16. Басейн $\mathcal{A}(1)$ для $z^3-z$}
Побудувати графічне зображення басейну притягання $\mathcal{A}(1)$ для
$f(z)=z^3-z$.

\begin{korotko}
Корені: $0$, $1$, $-1$. Ньютон: $z_{k+1}=\dfrac{2z_k^3}{3z_k^2-1}$;
точки з $|z_k-1|<\varepsilon$~--- $\mathcal{A}(1)$.
\end{korotko}

Для $f(z)=z^3-z$ корені $0$, $1$, $-1$; $f'(z)=3z^2-1$,
\begin{equation*}
  z_{k+1}=z_k-\frac{z_k^3-z_k}{3z_k^2-1}=\frac{2z_k^3}{3z_k^2-1}.
\end{equation*}

\textbf{Алгоритм побудови.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Розбити комплексну площину на сітку (наприклад, $[-1.5,1.5]^2$).
  \item Для кожного $z_0$ виконати ітерації $N(z)$.
  \item Якщо $|z_k-1|<\varepsilon$, зафарбувати піксель (колір може залежати від $k$).
  \item Інакше~--- інший колір (басейни $0$ та $-1$).
\end{enumerate}

\section{Фрактальна розмірність атрактора Лоренца}
% =====================================================================

Система Лоренца~--- класична система трьох нелінійних ОДР з хаотичною поведінкою:
\begin{align*}
  \dot{x} &= \sigma(y-x), &
  \dot{y} &= x(\rho-z)-y, &
  \dot{z} &= xy-\beta z.
\end{align*}
При $\sigma=10$, $\rho=28$, $\beta=8/3$ траєкторії сходяться до \textbf{дивного атрактора}
(рис.~\ref{fig:lorenz-attractor})~--- множини з нульовим фазовим об'ємом і дробовою
розмірністю (близько $2{,}06$).

\begin{figure}[htbp]
  \centering
  \includegraphics[width=0.72\linewidth]{lorenz_attractor.png}
  \caption{Дивний атрактор Лоренца у 3D-просторі ($\sigma=10$, $\rho=28$, $\beta=8/3$).}
  \label{fig:lorenz-attractor}
\end{figure}

Існують три основні підходи до обчислення (оцінки) фрактальної розмірності.

\subsection{Розмірність Ляпунова (гіпотеза Каплана--Йорке)}

\begin{korotko}
Інтегруємо систему та рівняння у варіаціях; за Грамом--Шмідтом~---
$\lambda_1,\lambda_2,\lambda_3$. Для Лоренца $D_L=2+\lambda_1/|\lambda_3|$.
\end{korotko}

\textbf{Обґрунтування.} Найточніший метод, коли відома модель. Фрактальна структура
виникає через «розтягування» і «згортання» фазового простору; це описують показники
Ляпунова $\lambda_1\ge\lambda_2\ge\lambda_3$. Для Лоренца:
$\lambda_1>0$, $\lambda_2=0$, $\lambda_3<0$; сума показників від'ємна (дисипативність).

\textbf{Алгоритм.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Одночасно інтегрувати систему Лоренца та рівняння у варіаціях (матрицю Якобі
        вздовж траєкторії).
  \item На кожному кроці ортогоналізувати (Грам--Шмідт) і обчислити
        $\lambda_1,\lambda_2,\lambda_3$.
  \item Розмірність за Капланом--Йорком:
        \[
          D_L=j+\frac{\sum_{i=1}^{j}\lambda_i}{|\lambda_{j+1}|},
        \]
        де $j$~--- максимальне ціле, для якого $\sum_{i=1}^{j}\lambda_i>0$.
        Для Лоренца $j=2$, тому
        \[
          D_L=2+\frac{\lambda_1+\lambda_2}{|\lambda_3|}
             =2+\frac{\lambda_1}{|\lambda_3|}.
        \]
\end{enumerate}

\subsection{Кореляційна розмірність (Грассбергер--Прокачча)}

\begin{korotko}
З траєкторії $X_1,\ldots,X_N$ будують $C(r)$~--- частку пар на відстані $<r$;
$D_2=\lim_{r\to0}\dfrac{\ln C(r)}{\ln r}$~--- нахил $\ln C$ vs $\ln r$.
\end{korotko}

\textbf{Обґрунтування.} Зручний для часових рядів; стійкий до шуму, ефективніший за
сіткові методи.

\textbf{Алгоритм.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Кореляційний інтеграл (частка пар точок на відстані меншій за $r$):
        \[
          C(r)=\lim_{N\to\infty}\frac{2}{N(N-1)}
          \sum_{i=1}^{N}\sum_{j=i+1}^{N}H(r-\|X_i-X_j\|),
        \]
        де $H$~--- функція Гевісайда.
  \item Для фракталів $C(r)\sim r^D$ при $r\to0$; будують графік $\ln C(r)$ від $\ln r$.
  \item Кореляційна розмірність:
        \[
          D_2=\lim_{r\to0}\frac{\ln C(r)}{\ln r}
        \]
        (нахил лінійної ділянки).
\end{enumerate}

\subsection{Ємнісна розмірність (box-counting)}

\begin{korotko}
Траєкторія в $\mathbb{R}^3$; рахують клітинки $N(r)$ сітки з кроком $r$;
$D_0=\lim_{r\to0}\dfrac{\ln N(r)}{\ln(1/r)}$. Дорого: $\mathcal{O}(1/r^3)$.
\end{korotko}

\textbf{Обґрунтування.} Пряме геометричне визначення; для 3D потребує великих ресурсів.

\textbf{Алгоритм.}
\begin{enumerate}[leftmargin=1.6em,itemsep=0.25em]
  \item Чисельно інтегрувати систему (наприклад, Рунге--Кутта), отримати довгу траєкторію.
  \item Розбити $\mathbb{R}^3$ на кубічну сітку з кроком $r$.
  \item Підрахувати $N(r)$~--- клітинки з хоча б однією точкою траєкторії.
  \item Зменшувати $r$, повторювати підрахунок.
  \item Ємнісна розмірність:
        \[
          D_0=\lim_{r\to0}\frac{\ln N(r)}{\ln(1/r)}.
        \]
\end{enumerate}
