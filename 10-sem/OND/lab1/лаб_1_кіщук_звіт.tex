\section{Завдання 3}
Задана система:
\[
\begin{cases}
\dot{x} = -2x - 5y, \\
\dot{y} = 2x + 2y.
\end{cases}
\]

\subsection{Аналітичний розв'язок}
Запишемо систему у матричному вигляді:
\[
X' = AX, \qquad
A =
\begin{pmatrix}
-2 & -5 \\
2 & 2
\end{pmatrix}.
\]

Характеристичне рівняння:
\[
\det(A-\lambda E)=
\begin{vmatrix}
-2-\lambda & -5\\
2 & 2-\lambda
\end{vmatrix}
=(-2-\lambda)(2-\lambda)-(-10)
=\lambda^2+6=0.
\]
Отже,
\[
\lambda_{1,2}=\pm i\sqrt{6}.
\]
Власні значення суто уявні, тому особлива точка $(0,0)$ є \textbf{центром}; система \textbf{стійка}, але не асимптотично.

Для $\lambda_1=i\sqrt{6}$:
\[
\begin{pmatrix}
-2-i\sqrt{6} & -5\\
2 & 2-i\sqrt{6}
\end{pmatrix}
\begin{pmatrix}
v_1\\v_2
\end{pmatrix}
=
\begin{pmatrix}
0\\0
\end{pmatrix}.
\]
Із другого рівняння:
\[
2v_1+(2-i\sqrt{6})v_2=0
\;\Rightarrow\;
v_1=\left(-1+i\frac{\sqrt{6}}{2}\right)v_2.
\]
Нехай $v_2=2$, тоді $v_1=-2+i\sqrt{6}$, тобто
\[
V=
\begin{pmatrix}
-2+i\sqrt{6}\\
2
\end{pmatrix}
=
\begin{pmatrix}
-2\\2
\end{pmatrix}
+i
\begin{pmatrix}
\sqrt{6}\\0
\end{pmatrix}.
\]

Комплексний розв'язок:
\[
Z(t)=e^{i\sqrt{6}t}V.
\]
Виділяючи дійсну та уявну частини, отримуємо фундаментальні дійсні розв'язки:
\[
X_1(t)=
\begin{pmatrix}
-2\cos(\sqrt{6}t)-\sqrt{6}\sin(\sqrt{6}t)\\
2\cos(\sqrt{6}t)
\end{pmatrix},
\]
\[
X_2(t)=
\begin{pmatrix}
\sqrt{6}\cos(\sqrt{6}t)-2\sin(\sqrt{6}t)\\
2\sin(\sqrt{6}t)
\end{pmatrix}.
\]
Тоді загальний дійсний розв'язок:
\[
\begin{cases}
x(t)=C_1\bigl(-2\cos(\sqrt{6}t)-\sqrt{6}\sin(\sqrt{6}t)\bigr)
+C_2\bigl(\sqrt{6}\cos(\sqrt{6}t)-2\sin(\sqrt{6}t)\bigr),\\
y(t)=2C_1\cos(\sqrt{6}t)+2C_2\sin(\sqrt{6}t).
\end{cases}
\]

\subsection{Якісний опис фазового портрета}
\begin{itemize}
    \item Тип особливої точки: центр у початку координат $(0,0)$.
    \item Траєкторії: замкнені криві (еліптичного типу), що оточують початок координат.
    \item Напрям руху: проти годинникової стрілки (наприклад, у точці $(1,0)$ маємо $\dot{x}=-2<0$, $\dot{y}=2>0$).
\end{itemize}

\section{Завдання 6}
Задана система:
\[
\begin{cases}
\dot{x} = x, \\
\dot{y} = 2x - y.
\end{cases}
\]

\subsection{Аналітичний розв'язок}
Матриця системи:
\[
A=
\begin{pmatrix}
1 & 0\\
2 & -1
\end{pmatrix}.
\]

Характеристичне рівняння:
\[
\det(A-\lambda E)=
\begin{vmatrix}
1-\lambda & 0\\
2 & -1-\lambda
\end{vmatrix}
=(1-\lambda)(-1-\lambda)=0.
\]
Отже,
\[
\lambda_1=1,\qquad \lambda_2=-1.
\]
Оскільки власні значення дійсні та різних знаків, особлива точка $(0,0)$ є \textbf{сідлом} (нестійка).

Власні вектори:

Для $\lambda_1=1$:
\[
\begin{pmatrix}
0 & 0\\
2 & -2
\end{pmatrix}
\begin{pmatrix}
v_1\\v_2
\end{pmatrix}
=
\begin{pmatrix}
0\\0
\end{pmatrix}
\Rightarrow v_1=v_2
\Rightarrow
V_1=
\begin{pmatrix}
1\\1
\end{pmatrix}.
\]

Для $\lambda_2=-1$:
\[
\begin{pmatrix}
2 & 0\\
2 & 0
\end{pmatrix}
\begin{pmatrix}
v_1\\v_2
\end{pmatrix}
=
\begin{pmatrix}
0\\0
\end{pmatrix}
\Rightarrow v_1=0
\Rightarrow
V_2=
\begin{pmatrix}
0\\1
\end{pmatrix}.
\]

Загальний розв'язок:
\[
\begin{pmatrix}
x(t)\\y(t)
\end{pmatrix}
=
C_1 e^t
\begin{pmatrix}
1\\1
\end{pmatrix}
+
C_2 e^{-t}
\begin{pmatrix}
0\\1
\end{pmatrix},
\]
тобто
\[
\begin{cases}
x(t)=C_1 e^t,\\
y(t)=C_1 e^t + C_2 e^{-t}.
\end{cases}
\]

\subsection{Якісний опис фазового портрета}
\begin{itemize}
    \item Тип особливої точки: сідло у точці $(0,0)$.
    \item Сепаратриси:
    \[
    y=x \quad (\lambda_1=1>0,\ \text{рух від початку координат}),
    \]
    \[
    x=0 \quad (\lambda_2=-1<0,\ \text{рух до початку координат}).
    \]
    \item Інші траєкторії мають гіперболічний характер між сепаратрисами.
\end{itemize}
