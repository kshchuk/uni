\section{Мета та завдання роботи}

\subsection{Мета}
Реалізувати класифікацію точки до одного з трьох типів за відстанню до центру мас класу, порівняти результати для метрик $L2$ (евклідова) та $L1$ (мангеттенська), а також дослідити граничні випадки.

\subsection{Умова для парного номера}
Для парного номера у списку використано змішані розподіли:
\begin{itemize}
    \item Тип 1 --- рівномірний;
    \item Тип 2 --- нормальний;
    \item Тип 3 --- рівномірний.
\end{itemize}

\subsection{Завдання}
\begin{enumerate}
    \item Згенерувати три навчальні множини точок та множину тестових точок.
    \item Для кожної навчальної множини обчислити центроїд.
    \item Для кожної тестової точки порахувати відстані до центроїдів за $L2$ та $L1$.
    \item Призначити клас за правилом мінімальної відстані.
    \item Порівняти результати $L2$ і $L1$ на тестовому наборі.
\end{enumerate}

\section{Теоретичні відомості}

Нехай маємо три класи точок $C_1, C_2, C_3$. Для кожного класу обчислюється центроїд:
\begin{equation}
\mu_k = \frac{1}{|C_k|} \sum_{x \in C_k} x, \quad k \in \{1,2,3\}.
\end{equation}

Для тестової точки $t=(t_x,t_y)$ відстані до центроїда $\mu_k=(\mu_{k,x},\mu_{k,y})$:
\begin{align}
d_{L2}(t,\mu_k) &= \sqrt{(t_x-\mu_{k,x})^2 + (t_y-\mu_{k,y})^2}, \\
d_{L1}(t,\mu_k) &= |t_x-\mu_{k,x}| + |t_y-\mu_{k,y}|.
\end{align}

Рішення класифікатора:
\begin{equation}
\hat{k} = \arg\min_k d(t,\mu_k).
\end{equation}

\section{Фрагмент коду класифікації}
\begin{lstlisting}[language=Python, basicstyle=\small]
def pairwise_distances(point, centroids):
    names = list(centroids.keys())
    centers = np.array([centroids[n] for n in names])
    diffs = centers - point
    d_l2 = np.sqrt(np.sum(diffs ** 2, axis=1))
    d_l1 = np.sum(np.abs(diffs), axis=1)
    return names, d_l2, d_l1
\end{lstlisting}
